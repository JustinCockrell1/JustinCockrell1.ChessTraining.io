Day 7 Hangman

							
def clearScreen(): 
print("\n"*10)

lives=5
guessedLetters = []
correctlyGuessed = []

print("Welcome to hangman!")
word = input("Enter word to guess")

gameRunning=True

currentGuess = "_ "*len(word)

while gameRunning:
clearScreen()

draw = f"""
____
| | You have {lives} lives left
|
| Guessed letters {str(guessedLetters)}
|
| {correctlyGuessed}
|
____|____

{currentGuess}
"""

print(draw)
letter = input("Guess a letter: ")
if(letter in word):
if(letter not in correctlyGuessed):
correctlyGuessed+=letter
elif(letter not in guessedLetters):
guessedLetters+=letter
lives-=1

currentGuess = ""
for char in word:
if char in correctlyGuessed:
currentGuess+=char+""
else:
currentGuess+="_ "

if(currentGuess==word):
print("Congratulations you won!")
gameRunning=False
if(lives==0):
print("You lost")
gameRunning=False
